Iterate Set

import java.util.HashSet;
import java.util.Set;

public void iterateSet() {
		Set<String> set = new HashSet<>();

		for (int i = 0; i < 10; i++) {
			set.add("value" + i);
		}

		set.stream().forEach(System.out::println);
	}
